### Variables
```ocaml
// Constants.
a: str : "text";

// Variables.
b: i32 = 29;

// Functions.
sum: (a: i32, b: i32) -> i32 = {
  return a + b;
}

// Structs.
Point2D: struct = {
  x: f32;
  y: f32;
}
```
